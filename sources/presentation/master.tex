%% Automatische Gliederungsfolien
% Overview page at the start of each day (section) - shows ONLY topics for that day
% Only show if section has subsections
\AtBeginSection[] {
  \ifnum\value{subsection}>0
  \begin{frame}<beamer>
    \thispagestyle{empty}
    \frametitle{Überblick \insertsectionhead}
    \vspace{-5mm}
    {\large
    \tableofcontents[currentsection,hideothersubsections,sectionstyle=show/hide]
    }
  \end{frame}
  \fi
}

% Overview page before each subtopic (subsection) - shows ONLY current section's topics
\AtBeginSubsection[] {
  \begin{frame}<beamer>
    \thispagestyle{empty}
    \frametitle{Überblick - \insertsubsectionhead}
    \vspace{-5mm}
    {\large
    \tableofcontents[currentsection,sectionstyle=show/hide,subsectionstyle=show/shaded/hide]
    }
  \end{frame}
}
